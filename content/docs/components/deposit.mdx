## Usage

Wrap your app with the Nexus provider first. See [Get Started](/docs/get-started) for provider setup.

```tsx showLineNumbers
import { NexusOne } from "@/components/nexus-one/nexus-one";
import { encodeFunctionData } from "viem";

// Configure these for your app.
const APP_DEPOSIT_CONTRACT = "0x...";
const DESTINATION_TOKEN = "0x...";
const DESTINATION_CHAIN_ID = 42161;
const DESTINATION_TOKEN_SYMBOL = "USDT";
const DESTINATION_TOKEN_DECIMALS = 6;

const APP_DEPOSIT_ABI = [
  {
    inputs: [
      { internalType: "address", name: "token", type: "address" },
      { internalType: "uint256", name: "amount", type: "uint256" },
      { internalType: "address", name: "user", type: "address" },
    ],
    name: "deposit",
    outputs: [],
    stateMutability: "nonpayable",
    type: "function",
  },
] as const;

function buildDepositExecuteConfig(
  tokenAddress: `0x${string}`,
  amount: bigint,
  user: `0x${string}`,
) {
  return {
    to: APP_DEPOSIT_CONTRACT,
    data: encodeFunctionData({
      abi: APP_DEPOSIT_ABI,
      functionName: "deposit",
      args: [tokenAddress, amount, user],
      // Aave reference: supply(asset, amount, onBehalfOf, referralCode)
    }),
    tokenApproval: {
      token: tokenAddress,
      amount,
      spender: APP_DEPOSIT_CONTRACT,
    },
  };
}

const depositConfig = {
  mode: "deposit",
  deposit: {
    // Configure the destination asset Nexus should deliver.
    chainId: DESTINATION_CHAIN_ID,
    tokenSymbol: DESTINATION_TOKEN_SYMBOL,
    tokenDecimals: DESTINATION_TOKEN_DECIMALS,
    tokenAddress: DESTINATION_TOKEN,

    // Configure display metadata for your app/protocol.
    title: "Your App",
    protocol: "Your App",

    // Configure the app call Nexus should execute after the swap settles.
    executeDeposit: (_symbol, tokenAddress, amount, _chainId, user) =>
      buildDepositExecuteConfig(tokenAddress, amount, user),
  },
} as const;

<NexusOne config={depositConfig} connectedAddress={address} />;
```

## Configuration

Deposit uses only `mode: "deposit"`. This mode is exact-out: users enter the destination deposit amount, pay-with sources are selected, and the flow submits through `swapAndExecute`.

| Prop | Required | Description |
| --- | --- | --- |
| `config.mode` | Yes | Must be `"deposit"`. |
| `config.deposit` | Yes | Single destination and app-call config. Deposit opens the form directly. |
| `config.deposit.executeDeposit` | Yes | Transaction builder passed to `swapAndExecute`. |
| `config.prefill.amount` | Optional | Prefills the exact output deposit amount. |
| `config.allowedSourcePairs` | Optional | Restricts pay-with source assets. |
| `connectedAddress` | Optional | Wallet address to use. If omitted, the connected wagmi account is used. |
| `embed` | Optional | Defaults to `true`. Set `false` to render as a modal surface. |
| `open`, `onOpenChange`, `defaultOpen` | Optional | Control the modal when `embed={false}`. |
| `onStart`, `onComplete`, `onError`, `onClose`, `onConnectWallet` | Optional | Host app callbacks. Use `onConnectWallet` to route the element CTA to your app-level wallet flow. |

## Modal Usage

Render `embed={false}` to invoke Deposit from a button and show the element as a modal.

```tsx showLineNumbers
<NexusOne
  config={depositConfig}
  connectedAddress={address}
  embed={false}
  defaultOpen={false}
/>;
```

Control the modal from app state when needed.

```tsx showLineNumbers
const [open, setOpen] = useState(false);
 
<NexusOne
  config={depositConfig}
  connectedAddress={address}
  embed={false}
  open={open}
  onOpenChange={setOpen}
/>
```

## Supporting Multiple Opportunities / Selector UI

Since `NexusOne` is optimized to execute a single, targeted deposit flow, it accepts a single `deposit` configuration block. If your application needs to display multiple opportunities (such as choosing between Aave, Compound, or other vaults), you can build a simple selector in your parent component and render `NexusOne` dynamically with the chosen opportunity:

```tsx showLineNumbers
import { useState } from "react";
import { NexusOne } from "@/components/nexus-one/nexus-one";
import { encodeFunctionData } from "viem";

// Define the Aave & Compound pools/ABIs...

const OPPORTUNITIES = {
  aave: {
    chainId: 42161,
    tokenSymbol: "USDT",
    tokenDecimals: 6,
    tokenAddress: "0xFd086bC7CD5C481DCC9C85ebE478A1C0b69FCbb9",
    title: "Aave Market",
    protocol: "Aave",
    executeDeposit: (symbol, tokenAddress, amount, chainId, user) => ({
      to: "0x794a61358D6845594F94dc1DB02A252b5b4814aD",
      data: encodeFunctionData({ /* Aave supply arguments */ }),
      tokenApproval: { token: tokenAddress, amount, spender: "0x794a61358D6845594F94dc1DB02A252b5b4814aD" }
    })
  },
  compound: {
    chainId: 8453,
    tokenSymbol: "USDC",
    tokenDecimals: 6,
    tokenAddress: "0x833589fCD6eDb6E08f4c7C32D4f71b54bdA02913",
    title: "Compound Market",
    protocol: "Compound",
    executeDeposit: (symbol, tokenAddress, amount, chainId, user) => ({
      to: "0xc3d688B66703497DAA19211EEdff47f25384cdc3",
      data: encodeFunctionData({ /* Compound supply arguments */ }),
      tokenApproval: { token: tokenAddress, amount, spender: "0xc3d688B66703497DAA19211EEdff47f25384cdc3" }
    })
  }
};

export function MultiDepositShowcase({ address }) {
  const [selectedOpt, setSelectedOpt] = useState<keyof typeof OPPORTUNITIES>("aave");

  return (
    <div className="space-y-4">
      <div className="flex flex-col gap-1.5 max-w-xs mx-auto">
        <label className="text-xs font-semibold text-muted-foreground uppercase tracking-wider">
          Select Opportunity
        </label>
        <select
          value={selectedOpt}
          onChange={(e) => setSelectedOpt(e.target.value as keyof typeof OPPORTUNITIES)}
          className="w-full h-10 px-3 rounded-lg border border-border bg-background text-sm focus:outline-none focus:ring-2 focus:ring-primary focus:border-transparent cursor-pointer"
        >
          <option value="aave">Aave (USDT on Arbitrum)</option>
          <option value="compound">Compound (USDC on Base)</option>
        </select>
      </div>

      <NexusOne
        config={{
          mode: "deposit",
          deposit: OPPORTUNITIES[selectedOpt]
        }}
        connectedAddress={address}
      />
    </div>
  );
}
```
