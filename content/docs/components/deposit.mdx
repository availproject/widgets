## Usage

Wrap your app with the Nexus provider first. See [Get Started](/docs/get-started) for provider setup.

```tsx showLineNumbers
import { NexusOne } from "@/components/nexus-one/nexus-one";
import { encodeFunctionData } from "viem";

const AAVE_POOL_ARBITRUM = "0x794a61358D6845594F94dc1DB02A252b5b4814aD";
const USDT_ARBITRUM = "0xFd086bC7CD5C481DCC9C85ebE478A1C0b69FCbb9";

const AAVE_ABI = [
  {
    inputs: [
      { internalType: "address", name: "asset", type: "address" },
      { internalType: "uint256", name: "amount", type: "uint256" },
      { internalType: "address", name: "onBehalfOf", type: "address" },
      { internalType: "uint16", name: "referralCode", type: "uint16" },
    ],
    name: "supply",
    outputs: [],
    stateMutability: "nonpayable",
    type: "function",
  },
] as const;

export function DepositExample({ address }: { address?: `0x${string}` }) {
  return (
    <NexusOne
      connectedAddress={address}
      config={{
        mode: "deposit",
        prefill: {
          amount: "0.1",
        },
        opportunities: [
          {
            id: "aave-arbitrum-usdt",
            title: "Aave",
            protocol: "Aave",
            subtitle: "Deposit USDT on Aave on Arbitrum",
            logo: "https://files.availproject.org/uploads/2026-04-16/aave.svg",
            chainId: 42161,
            tokenSymbol: "USDT",
            tokenAddress: USDT_ARBITRUM,
            tokenLogo:
              "https://raw.githubusercontent.com/availproject/nexus-assets/refs/heads/main/tokens/usdt/logo.png",
            execute: (amount, connectedAddress) => ({
              to: AAVE_POOL_ARBITRUM,
              data: encodeFunctionData({
                abi: AAVE_ABI,
                functionName: "supply",
                args: [USDT_ARBITRUM, amount, connectedAddress, 0],
              }),
              gas: 300000n,
              tokenApproval: {
                token: USDT_ARBITRUM,
                amount,
                spender: AAVE_POOL_ARBITRUM,
              },
            }),
          },
        ],
      }}
    />
  );
}
```

## Configuration

Deposit uses only `mode: "deposit"`. This mode is exact-out: users enter the destination deposit amount, pay-with sources are selected, and the flow submits through `swapAndExecute`.

| Prop | Required | Description |
| --- | --- | --- |
| `config.mode` | Yes | Must be `"deposit"`. |
| `config.opportunities` | Yes | Must contain at least one opportunity. One opportunity opens the deposit form directly. Multiple opportunities render the opportunity picker first. |
| `config.opportunities[].execute` | Yes | Transaction builder passed to `swapAndExecute`. |
| `config.prefill.amount` | Optional | Prefills the exact output deposit amount. |
| `config.allowedSourcePairs` | Optional | Restricts pay-with source assets. |
| `connectedAddress` | Optional | Wallet address to use. If omitted, the connected wagmi account is used. |
| `embed` | Optional | Defaults to `true`. Set `false` to render as a modal surface. |
| `onStart`, `onComplete`, `onError`, `onClose` | Optional | Lifecycle callbacks for host apps. |

```ts title="DepositConfig"
interface DepositConfig {
  mode: "deposit";
  prefill?: {
    amount?: string;
  };
  opportunities: DepositOpportunity[];
  allowedSourcePairs?: {
    token: `0x${string}`;
    chain: number;
  }[];
}

interface DepositOpportunity {
  id: string;
  protocol: string;
  title?: string;
  subtitle?: string;
  logo?: string;
  chainId: number;
  tokenSymbol: string;
  tokenLogo?: string;
  tokenAddress: `0x${string}`;
  execute:
    | ExecuteParams
    | ((amount: bigint, connectedAddress: `0x${string}`) => ExecuteParams);
}

type ExecuteParams = {
  to: `0x${string}`;
  data?: `0x${string}`;
  value?: bigint;
  gas: bigint;
  tokenApproval?: {
    token: `0x${string}`;
    amount: bigint;
    spender: `0x${string}`;
  };
};
```
